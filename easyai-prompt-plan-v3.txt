---- STEP 1: httpd.conf ----
File: C:\xampp\apache\conf\httpd.conf
Uncomment: Include conf/extra/httpd-vhosts.conf

---- STEP 2: httpd-vhosts.conf ----
File: C:\xampp\apache\conf\extra\httpd-vhosts.conf
Add:

<VirtualHost *:80>
    ServerName easyai.local
    DocumentRoot "C:/xampp/htdocs/easyai/public"
    <Directory "C:/xampp/htdocs/easyai/public">
        AllowOverride All
        Require all granted
    </Directory>
    ErrorLog "C:/xampp/apache/logs/easyai-error.log"
</VirtualHost>

<VirtualHost *:80>
    ServerName pis.easyai.local
    DocumentRoot "C:/xampp/htdocs/easyai/public"
    <Directory "C:/xampp/htdocs/easyai/public">
        AllowOverride All
        Require all granted
    </Directory>
    ErrorLog "C:/xampp/apache/logs/pis-error.log"
</VirtualHost>

---- STEP 3: hosts file ----
File: C:\Windows\System32\drivers\etc\hosts (open as Administrator)
Add:
127.0.0.1    easyai.local
127.0.0.1    pis.easyai.local

---- STEP 4: Restart Apache in XAMPP ----
Test: http://easyai.local and http://pis.easyai.local both load Laravel
